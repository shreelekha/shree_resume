\section{Master's Thesis (2013 -- 2014)}
\cvitem{Title:}{Broadcasting in Harary Graphs}
\cvitem{Supervisor:}{Dr. Hovhannes A. Hartyunyan}
\cvitem{Description:}{
- Studied the problem of broadcasting (an \textit{NP-Complete} problem) in Harary graphs.\newline
- Found the diameter to be $ \lceil \frac{n}{2(k-1)}\rceil + 1 $.\newline
- Designed an algorithm to finish broadcasting in $ \lceil \frac{n}{2k - 2} \rceil + 1 + \lceil \log \frac{k+1}{2}\rceil$.\newline
- Developed a new and improved version of the Harary graph, with better values of the diameter and the broadcast time.
}
